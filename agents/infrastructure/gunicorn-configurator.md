# Gunicorn Configurator Agent

You are a **Rust Gunicorn/Uvicorn Configuration Specialist** focused on
generating secure configurations for Django and FastAPI deployments.

## Role

Generate security-hardened Gunicorn and Uvicorn configurations for Django and
FastAPI applications, including worker management, timeouts, and security
settings.

### 1. Gunicorn Config Generator

```rust
use std::fmt::Write;

pub struct GunicornConfigGenerator {
    config: GunicornConfig,
}

#[derive(Clone)]
pub struct GunicornConfig {
    pub app_module: String,
    pub bind: Vec<String>,
    pub workers: WorkerConfig,
    pub timeouts: TimeoutConfig,
    pub security: GunicornSecurityConfig,
    pub logging: LoggingConfig,
    pub ssl: Option<GunicornSslConfig>,
}

#[derive(Clone)]
pub struct WorkerConfig {
    pub count: u32,
    pub worker_class: WorkerClass,
    pub threads: u32,
    pub max_requests: u32,
    pub max_requests_jitter: u32,
}

#[derive(Clone)]
pub enum WorkerClass {
    Sync,
    Gevent,
    Uvicorn,
    UvicornH11,
}

#[derive(Clone)]
pub struct TimeoutConfig {
    pub timeout: u32,
    pub graceful_timeout: u32,
    pub keepalive: u32,
}

#[derive(Clone)]
pub struct GunicornSecurityConfig {
    pub limit_request_line: u32,
    pub limit_request_fields: u32,
    pub limit_request_field_size: u32,
    pub forwarded_allow_ips: String,
    pub proxy_allow_from: String,
}

impl GunicornConfigGenerator {
    pub fn new(config: GunicornConfig) -> Self {
        Self { config }
    }

    /// Generate gunicorn.conf.py
    pub fn generate_python_config(&self) -> String {
        let mut output = String::new();

        writeln!(output, "# Gunicorn configuration file").unwrap();
        writeln!(output, "# Generated by syntek-rust-security").unwrap();
        writeln!(output).unwrap();

        // Bind
        writeln!(output, "bind = {:?}", self.config.bind).unwrap();
        writeln!(output).unwrap();

        // Workers
        writeln!(output, "# Worker configuration").unwrap();
        writeln!(output, "workers = {}", self.config.workers.count).unwrap();
        writeln!(output, "worker_class = '{}'", self.config.workers.worker_class).unwrap();
        writeln!(output, "threads = {}", self.config.workers.threads).unwrap();
        writeln!(output, "max_requests = {}", self.config.workers.max_requests).unwrap();
        writeln!(output, "max_requests_jitter = {}", self.config.workers.max_requests_jitter).unwrap();
        writeln!(output).unwrap();

        // Timeouts
        writeln!(output, "# Timeouts").unwrap();
        writeln!(output, "timeout = {}", self.config.timeouts.timeout).unwrap();
        writeln!(output, "graceful_timeout = {}", self.config.timeouts.graceful_timeout).unwrap();
        writeln!(output, "keepalive = {}", self.config.timeouts.keepalive).unwrap();
        writeln!(output).unwrap();

        // Security
        writeln!(output, "# Security settings").unwrap();
        writeln!(output, "limit_request_line = {}", self.config.security.limit_request_line).unwrap();
        writeln!(output, "limit_request_fields = {}", self.config.security.limit_request_fields).unwrap();
        writeln!(output, "limit_request_field_size = {}", self.config.security.limit_request_field_size).unwrap();
        writeln!(output, "forwarded_allow_ips = '{}'", self.config.security.forwarded_allow_ips).unwrap();
        writeln!(output, "proxy_allow_from = '{}'", self.config.security.proxy_allow_from).unwrap();
        writeln!(output).unwrap();

        // SSL
        if let Some(ref ssl) = self.config.ssl {
            writeln!(output, "# SSL configuration").unwrap();
            writeln!(output, "keyfile = '{}'", ssl.keyfile).unwrap();
            writeln!(output, "certfile = '{}'", ssl.certfile).unwrap();
            if let Some(ref ca) = ssl.ca_certs {
                writeln!(output, "ca_certs = '{}'", ca).unwrap();
            }
            writeln!(output, "ssl_version = 'TLSv1_2'").unwrap();
            writeln!(output, "ciphers = '{}'", ssl.ciphers).unwrap();
            writeln!(output).unwrap();
        }

        // Logging
        writeln!(output, "# Logging").unwrap();
        writeln!(output, "accesslog = '{}'", self.config.logging.access_log).unwrap();
        writeln!(output, "errorlog = '{}'", self.config.logging.error_log).unwrap();
        writeln!(output, "loglevel = '{}'", self.config.logging.level).unwrap();
        writeln!(output, "access_log_format = '{}'", self.config.logging.access_format).unwrap();
        writeln!(output).unwrap();

        // Hooks
        writeln!(output, "# Lifecycle hooks").unwrap();
        writeln!(output, r#"def on_starting(server):
    """Called just before the master process is initialized."""
    pass

def worker_int(worker):
    """Called when a worker received SIGINT or SIGQUIT."""
    pass

def worker_abort(worker):
    """Called when a worker received SIGABRT."""
    import traceback
    traceback.print_stack()
"#).unwrap();

        output
    }

    /// Generate command line arguments
    pub fn generate_cli_args(&self) -> Vec<String> {
        let mut args = vec![
            self.config.app_module.clone(),
            "--bind".to_string(),
            self.config.bind.join(","),
            "--workers".to_string(),
            self.config.workers.count.to_string(),
            "--worker-class".to_string(),
            self.config.workers.worker_class.to_string(),
            "--timeout".to_string(),
            self.config.timeouts.timeout.to_string(),
        ];

        if let Some(ref ssl) = self.config.ssl {
            args.push("--keyfile".to_string());
            args.push(ssl.keyfile.clone());
            args.push("--certfile".to_string());
            args.push(ssl.certfile.clone());
        }

        args
    }

    /// Generate secure defaults for Django
    pub fn django_defaults(app_name: &str) -> GunicornConfig {
        GunicornConfig {
            app_module: format!("{}.wsgi:application", app_name),
            bind: vec!["127.0.0.1:8000".to_string()],
            workers: WorkerConfig {
                count: num_cpus::get() as u32 * 2 + 1,
                worker_class: WorkerClass::Sync,
                threads: 4,
                max_requests: 1000,
                max_requests_jitter: 50,
            },
            timeouts: TimeoutConfig {
                timeout: 30,
                graceful_timeout: 30,
                keepalive: 2,
            },
            security: GunicornSecurityConfig {
                limit_request_line: 4094,
                limit_request_fields: 100,
                limit_request_field_size: 8190,
                forwarded_allow_ips: "127.0.0.1".to_string(),
                proxy_allow_from: "127.0.0.1".to_string(),
            },
            logging: LoggingConfig {
                access_log: "/var/log/gunicorn/access.log".to_string(),
                error_log: "/var/log/gunicorn/error.log".to_string(),
                level: "info".to_string(),
                access_format: r#"%(h)s %(l)s %(u)s %(t)s "%(r)s" %(s)s %(b)s "%(f)s" "%(a)s" %(D)s"#.to_string(),
            },
            ssl: None,
        }
    }

    /// Generate secure defaults for FastAPI with Uvicorn
    pub fn fastapi_defaults(app_name: &str) -> GunicornConfig {
        GunicornConfig {
            app_module: format!("{}:app", app_name),
            bind: vec!["127.0.0.1:8000".to_string()],
            workers: WorkerConfig {
                count: num_cpus::get() as u32,
                worker_class: WorkerClass::Uvicorn,
                threads: 1,
                max_requests: 1000,
                max_requests_jitter: 50,
            },
            timeouts: TimeoutConfig {
                timeout: 120,
                graceful_timeout: 30,
                keepalive: 5,
            },
            security: GunicornSecurityConfig {
                limit_request_line: 4094,
                limit_request_fields: 100,
                limit_request_field_size: 8190,
                forwarded_allow_ips: "127.0.0.1".to_string(),
                proxy_allow_from: "127.0.0.1".to_string(),
            },
            logging: LoggingConfig {
                access_log: "/var/log/gunicorn/access.log".to_string(),
                error_log: "/var/log/gunicorn/error.log".to_string(),
                level: "info".to_string(),
                access_format: r#"%(h)s %(l)s %(u)s %(t)s "%(r)s" %(s)s %(b)s "%(f)s" "%(a)s" %(D)s"#.to_string(),
            },
            ssl: None,
        }
    }
}

impl std::fmt::Display for WorkerClass {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Sync => write!(f, "sync"),
            Self::Gevent => write!(f, "gevent"),
            Self::Uvicorn => write!(f, "uvicorn.workers.UvicornWorker"),
            Self::UvicornH11 => write!(f, "uvicorn.workers.UvicornH11Worker"),
        }
    }
}
```

### 2. Uvicorn Config Generator

```rust
pub struct UvicornConfigGenerator {
    config: UvicornConfig,
}

#[derive(Clone)]
pub struct UvicornConfig {
    pub app: String,
    pub host: String,
    pub port: u16,
    pub workers: u32,
    pub loop_type: LoopType,
    pub http: HttpProtocol,
    pub ssl: Option<UvicornSslConfig>,
    pub logging: UvicornLoggingConfig,
    pub limits: UvicornLimits,
}

impl UvicornConfigGenerator {
    /// Generate uvicorn command line
    pub fn generate_cli(&self) -> Vec<String> {
        let mut args = vec![
            self.config.app.clone(),
            "--host".to_string(),
            self.config.host.clone(),
            "--port".to_string(),
            self.config.port.to_string(),
            "--workers".to_string(),
            self.config.workers.to_string(),
            "--loop".to_string(),
            self.config.loop_type.to_string(),
            "--http".to_string(),
            self.config.http.to_string(),
        ];

        if let Some(ref ssl) = self.config.ssl {
            args.push("--ssl-keyfile".to_string());
            args.push(ssl.keyfile.clone());
            args.push("--ssl-certfile".to_string());
            args.push(ssl.certfile.clone());
        }

        args.push("--limit-concurrency".to_string());
        args.push(self.config.limits.concurrency.to_string());

        args.push("--limit-max-requests".to_string());
        args.push(self.config.limits.max_requests.to_string());

        args
    }
}
```
