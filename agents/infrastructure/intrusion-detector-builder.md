# Intrusion Detector Builder Agent

## Role

Sonnet-based agent for building Rust intrusion detection and prevention systems
(IDS/IPS) with Snort/Suricata-compatible rule processing, network traffic
analysis, alert generation, and automated response capabilities.

/// IDS/IPS configuration
#[derive(Debug, Clone)]
pub struct IdsConfig {
    pub mode: IdsMode,
    pub rules_dirs: Vec<std::path::PathBuf>,
    pub home_net: Vec<IpNetwork>,
    pub external_net: Vec<IpNetwork>,
    pub alert_log: std::path::PathBuf,

}

pub enum IdsMode {

    Prevention,  // IPS - Alert and block
}

/// IP network representation
#[derive(Debug, Clone)]
pub struct IpNetwork {
    pub addr: IpAddr,
    pub prefix_len: u8,
}

impl IpNetwork {
    pub fn contains(&self, ip: &IpAddr) -> bool {
        match (self.addr, ip) {
            (IpAddr::V4(net), IpAddr::V4(test)) => {
                let net_bits = u32::from(net);
                let test_bits = u32::from(*test);
                let mask = !0u32 << (32 - self.prefix_len);
                (net_bits & mask) == (test_bits & mask)
            }
            (IpAddr::V6(net), IpAddr::V6(test)) => {
                let net_bits = u128::from(net);
                let test_bits = u128::from(*test);
                let mask = !0u128 << (128 - self.prefix_len);
                (net_bits & mask) == (test_bits & mask)
            }
            _ => false,
        }
    }
}

/// Main IDS/IPS engine
pub struct IntrusionDetector {
    config: IdsConfig,
    rules: Arc<RwLock<RuleEngine>>,
    alert_manager: Arc<AlertManager>,
    flow_tracker: Arc<RwLock<FlowTracker>>,
    blocker: Arc<RwLock<Blocker>>,
    pcap_logger: Option<Arc<PcapLogger>>,
    stats: Arc<RwLock<IdsStats>>,
}

impl IntrusionDetector {
    pub async fn new(config: IdsConfig) -> Result<Self, IdsError> {
        let rules = RuleEngine::load(&config.rules_dirs, &config).await?;
        let alert_manager = AlertManager::new(
            &config.alert_log,
            config.max_alerts_per_second,
        ).await?;
        let flow_tracker = FlowTracker::new();
        let blocker = Blocker::new(
            config.block_threshold,
            config.block_duration_seconds,
        );

        let pcap_logger = if config.enable_pcap_logging {
            Some(Arc::new(PcapLogger::new(&config.pcap_log_dir).await?))
        } else {
            None
        };

        Ok(Self {
            config,
            rules: Arc::new(RwLock::new(rules)),
            alert_manager: Arc::new(alert_manager),
            flow_tracker: Arc::new(RwLock::new(flow_tracker)),
            blocker: Arc::new(RwLock::new(blocker)),
            pcap_logger,
            stats: Arc::new(RwLock::new(IdsStats::default())),
        })
    }

    /// Analyze a packet and return any alerts
    pub async fn analyze_packet(&self, packet: &Packet) -> Result<Vec<Alert>, IdsError> {
        // Update stats
        {
            let mut stats = self.stats.write().await;
            stats.packets_analyzed += 1;
        }

        // Check if source is blocked
        if self.config.mode == IdsMode::Prevention {
            let blocker = self.blocker.read().await;
            if blocker.is_blocked(&packet.src_ip) {
                let mut stats = self.stats.write().await;
                stats.packets_blocked += 1;
                return Ok(vec![]);
            }
        }

        // Update flow tracking
        {
            let mut flow_tracker = self.flow_tracker.write().await;
            flow_tracker.update(packet);
        }

        // Run rules against packet
        let rules = self.rules.read().await;
        let matching_rules = rules.match_packet(packet, &self.config).await;

        let mut alerts = Vec::new();

        for rule in matching_rules {
            let action = if self.config.mode == IdsMode::Prevention {
                self.determine_action(&rule).await
            } else {
                AlertAction::Alert
            };

            let alert = Alert {
                id: self.generate_alert_id().await,
                timestamp: chrono::Utc::now(),
                rule_id: rule.sid,
                rule_name: rule.msg.clone(),
                severity: rule.severity,
                classification: rule.classtype.clone(),
                src_ip: packet.src_ip,
                dst_ip: packet.dst_ip,
                src_port: packet.src_port,
                dst_port: packet.dst_port,
                protocol: packet.protocol,
                message: rule.msg.clone(),
                payload_excerpt: self.extract_payload_excerpt(&packet.payload),
                action_taken: action,
            };

            // Handle action in IPS mode
            if self.config.mode == IdsMode::Prevention {
                match action {
                    AlertAction::Block => {
                        let mut blocker = self.blocker.write().await;
                        blocker.add_block(&packet.src_ip);
                    }
                    AlertAction::RateLimit => {
                        // Implement rate limiting logic
                    }
                    _ => {}
                }
            }

            // Log PCAP if enabled
            if let Some(ref logger) = self.pcap_logger {
                logger.log_packet(packet, &alert).await?;
            }

            // Record alert
            self.alert_manager.record_alert(&alert).await?;

            alerts.push(alert);
        }

        // Update alert stats
        {
            let mut stats = self.stats.write().await;
            stats.alerts_generated += alerts.len() as u64;
        }

        Ok(alerts)
    }

    async fn determine_action(&self, rule: &Rule) -> AlertAction {
        match rule.action {
            RuleAction::Alert => AlertAction::Alert,
            RuleAction::Drop | RuleAction::Reject => AlertAction::Block,
            RuleAction::Pass => AlertAction::Log,
        }
    }

    async fn generate_alert_id(&self) -> u64 {
        let mut stats = self.stats.write().await;
        stats.alert_counter += 1;
        stats.alert_counter
    }

    fn extract_payload_excerpt(&self, payload: &[u8]) -> String {
        let excerpt_len = payload.len().min(100);
        let excerpt = &payload[..excerpt_len];

        // Convert to printable string, replacing non-printable chars
        excerpt.iter()
            .map(|&b| if b.is_ascii_graphic() || b == b' ' { b as char } else { '.' })
            .collect()
    }

    /// Reload rules from disk
    pub async fn reload_rules(&self) -> Result<(), IdsError> {
        let new_rules = RuleEngine::load(&self.config.rules_dirs, &self.config).await?;
        let mut rules = self.rules.write().await;
        *rules = new_rules;
        tracing::info!("Rules reloaded");
        Ok(())
    }

    pub async fn stats(&self) -> IdsStats {
        self.stats.read().await.clone()
    }
}

    pub packets_blocked: u64,
    pub alerts_generated: u64,
    pub alert_counter: u64,
    pub rules_loaded: usize,
    pub active_blocks: usize,
}

/// Rule parser for Snort/Suricata format
pub struct RuleParser;

impl RuleParser {
    pub fn parse(rule_text: &str) -> Result<Rule, IdsError> {
        let rule_text = rule_text.trim();

        // Skip comments and empty lines
        if rule_text.is_empty() || rule_text.starts_with('#') {
            return Err(IdsError::SkippedLine);
        }

        // Parse header: action protocol src_addr src_port direction dst_addr dst_port
        let header_re = Regex::new(
            r"^(\w+)\s+(\w+)\s+(\S+)\s+(\S+)\s+(<>|->)\s+(\S+)\s+(\S+)\s*\((.*)\)\s*$"
        ).unwrap();

        let caps = header_re.captures(rule_text)
            .ok_or_else(|| IdsError::ParseError("Invalid rule format".to_string()))?;

        let action = Self::parse_action(caps.get(1).unwrap().as_str())?;
        let protocol = Self::parse_protocol(caps.get(2).unwrap().as_str())?;
        let src_addr = Self::parse_address(caps.get(3).unwrap().as_str())?;
        let src_port = Self::parse_port(caps.get(4).unwrap().as_str())?;
        let direction = Self::parse_direction(caps.get(5).unwrap().as_str())?;
        let dst_addr = Self::parse_address(caps.get(6).unwrap().as_str())?;
        let dst_port = Self::parse_port(caps.get(7).unwrap().as_str())?;
        let options_str = caps.get(8).unwrap().as_str();

        // Parse options
        let options = Self::parse_options(options_str)?;

        let sid = options.get("sid")
            .and_then(|s| s.parse().ok())
            .unwrap_or(0);
        let rev = options.get("rev")
            .and_then(|s| s.parse().ok())
            .unwrap_or(1);
        let msg = options.get("msg")
            .map(|s| s.trim_matches('"').to_string())
            .unwrap_or_default();
        let classtype = options.get("classtype")
            .map(|s| s.to_string())
            .unwrap_or_else(|| "unknown".to_string());

        let severity = Self::classtype_to_severity(&classtype);
        let content_matches = Self::parse_content_matches(&options);
        let pcre_matches = Self::parse_pcre_matches(&options)?;
        let flow = Self::parse_flow(&options);
        let flags = Self::parse_tcp_flags(&options);

        Ok(Rule {
            sid,
            rev,
            action,
            protocol,
            src_addr,
            src_port,
            direction,
            dst_addr,
            dst_port,
            msg,
            content_matches,
            pcre_matches,
            flow,
            flags,
            classtype,
            severity,
            metadata: options,
        })
    }

    fn parse_action(s: &str) -> Result<RuleAction, IdsError> {
        match s.to_lowercase().as_str() {
            "alert" => Ok(RuleAction::Alert),
            "drop" => Ok(RuleAction::Drop),
            "reject" => Ok(RuleAction::Reject),
            "pass" => Ok(RuleAction::Pass),
            "log" => Ok(RuleAction::Log),
            _ => Err(IdsError::ParseError(format!("Unknown action: {}", s))),
        }
    }

    fn parse_protocol(s: &str) -> Result<RuleProtocol, IdsError> {
        match s.to_lowercase().as_str() {
            "tcp" => Ok(RuleProtocol::TCP),
            "udp" => Ok(RuleProtocol::UDP),
            "icmp" => Ok(RuleProtocol::ICMP),
            "ip" => Ok(RuleProtocol::IP),
            "http" => Ok(RuleProtocol::HTTP),
            "dns" => Ok(RuleProtocol::DNS),
            "tls" | "ssl" => Ok(RuleProtocol::TLS),
            "any" => Ok(RuleProtocol::Any),
            _ => Err(IdsError::ParseError(format!("Unknown protocol: {}", s))),
        }
    }

    fn parse_address(s: &str) -> Result<AddressSpec, IdsError> {
        let s = s.trim();

        if s == "any" {
            return Ok(AddressSpec::Any);
        }
        if s == "$HOME_NET" {
            return Ok(AddressSpec::HomeNet);
        }
        if s == "$EXTERNAL_NET" || s == "!$HOME_NET" {
            return Ok(AddressSpec::ExternalNet);
        }
        if s.starts_with('!') {
            let inner = Self::parse_address(&s[1..])?;
            return Ok(AddressSpec::Negated(Box::new(inner)));
        }
        if s.starts_with('[') && s.ends_with(']') {
            let inner = &s[1..s.len()-1];
            let addrs: Result<Vec<_>, _> = inner.split(',')
                .map(|a| Self::parse_network(a.trim()))
                .collect();
            return Ok(AddressSpec::AddressList(addrs?));
        }

        Ok(AddressSpec::Address(Self::parse_network(s)?))
    }

    fn parse_network(s: &str) -> Result<IpNetwork, IdsError> {
        if let Some((addr, prefix)) = s.split_once('/') {
            let addr: IpAddr = addr.parse()
                .map_err(|_| IdsError::ParseError(format!("Invalid IP: {}", addr)))?;
            let prefix_len: u8 = prefix.parse()
                .map_err(|_| IdsError::ParseError(format!("Invalid prefix: {}", prefix)))?;
            Ok(IpNetwork { addr, prefix_len })
        } else {
            let addr: IpAddr = s.parse()
                .map_err(|_| IdsError::ParseError(format!("Invalid IP: {}", s)))?;
            let prefix_len = if addr.is_ipv4() { 32 } else { 128 };
            Ok(IpNetwork { addr, prefix_len })
        }
    }

    fn parse_port(s: &str) -> Result<PortSpec, IdsError> {
        let s = s.trim();

        if s == "any" {
            return Ok(PortSpec::Any);
        }
        if s.starts_with('!') {
            let inner = Self::parse_port(&s[1..])?;
            return Ok(PortSpec::Negated(Box::new(inner)));
        }
        if s.contains(':') {
            let parts: Vec<&str> = s.split(':').collect();
            let start = if parts[0].is_empty() { 0 } else { parts[0].parse()? };
            let end = if parts[1].is_empty() { 65535 } else { parts[1].parse()? };
            return Ok(PortSpec::Range(start, end));
        }
        if s.starts_with('[') && s.ends_with(']') {
            let inner = &s[1..s.len()-1];
            let ports: Result<Vec<u16>, _> = inner.split(',')
                .map(|p| p.trim().parse())
                .collect();
            return Ok(PortSpec::List(ports?));
        }

        Ok(PortSpec::Single(s.parse()?))
    }

    fn parse_direction(s: &str) -> Result<Direction, IdsError> {
        match s {
            "->" => Ok(Direction::Unidirectional),
            "<>" => Ok(Direction::Bidirectional),
            _ => Err(IdsError::ParseError(format!("Invalid direction: {}", s))),
        }
    }

    fn parse_options(options_str: &str) -> Result<HashMap<String, String>, IdsError> {
        let mut options = HashMap::new();

        // Split by semicolons, handling quoted strings
        let mut current = String::new();
        let mut in_quotes = false;

        for c in options_str.chars() {
            match c {
                '"' => {
                    in_quotes = !in_quotes;
                    current.push(c);
                }
                ';' if !in_quotes => {
                    let opt = current.trim();
                    if !opt.is_empty() {
                        if let Some((key, value)) = opt.split_once(':') {
                            options.insert(key.trim().to_string(), value.trim().to_string());
                        } else {
                            options.insert(opt.to_string(), String::new());
                        }
                    }
                    current.clear();
                }
                _ => current.push(c),
            }
        }

        // Don't forget the last option
        let opt = current.trim();
        if !opt.is_empty() {
            if let Some((key, value)) = opt.split_once(':') {
                options.insert(key.trim().to_string(), value.trim().to_string());
            }
        }

        Ok(options)
    }

    fn parse_content_matches(options: &HashMap<String, String>) -> Vec<ContentMatch> {
        let mut matches = Vec::new();

        // This is simplified - real implementation would track option order
        if let Some(content) = options.get("content") {
            let pattern = Self::parse_content_pattern(content);
            let nocase = options.contains_key("nocase");

            matches.push(ContentMatch {
                pattern,
                nocase,
                offset: options.get("offset").and_then(|s| s.parse().ok()),
                depth: options.get("depth").and_then(|s| s.parse().ok()),
                distance: options.get("distance").and_then(|s| s.parse().ok()),
                within: options.get("within").and_then(|s| s.parse().ok()),
                fast_pattern: options.contains_key("fast_pattern"),
                negated: content.starts_with('!'),
            });
        }

        matches
    }

    fn parse_content_pattern(content: &str) -> Vec<u8> {
        let content = content.trim_matches('"');
        let content = if content.starts_with('!') { &content[1..] } else { content };
        let content = content.trim_matches('"');

        let mut result = Vec::new();
        let mut chars = content.chars().peekable();

        while let Some(c) = chars.next() {
            if c == '|' {
                // Hex sequence
                let mut hex = String::new();
                while let Some(&next) = chars.peek() {
                    if next == '|' {
                        chars.next();
                        break;
                    }
                    if !next.is_whitespace() {
                        hex.push(next);
                    }
                    chars.next();
                }
                // Parse hex pairs
                for chunk in hex.as_bytes().chunks(2) {
                    if chunk.len() == 2 {
                        if let Ok(byte) = u8::from_str_radix(
                            std::str::from_utf8(chunk).unwrap_or("00"),
                            16
                        ) {
                            result.push(byte);
                        }
                    }
                }
            } else {
                result.push(c as u8);
            }
        }

        result
    }

    fn parse_pcre_matches(options: &HashMap<String, String>) -> Result<Vec<PcreMatch>, IdsError> {
        let mut matches = Vec::new();

        if let Some(pcre) = options.get("pcre") {
            // Format: "/pattern/modifiers"
            let pcre = pcre.trim_matches('"');
            let negated = pcre.starts_with('!');
            let pcre = if negated { &pcre[1..] } else { pcre };

            if pcre.starts_with('/') {
                if let Some(last_slash) = pcre.rfind('/') {
                    let pattern = &pcre[1..last_slash];
                    let modifiers = &pcre[last_slash+1..];

                    let regex = Regex::new(pattern)
                        .map_err(|e| IdsError::ParseError(format!("Invalid PCRE: {}", e)))?;

                    matches.push(PcreMatch {
                        pattern: regex,
                        modifiers: modifiers.to_string(),
                        negated,
                    });
                }
            }
        }

        Ok(matches)
    }

    fn parse_flow(options: &HashMap<String, String>) -> Option<FlowSpec> {
        options.get("flow").map(|flow| {
            FlowSpec {
                to_server: flow.contains("to_server"),
                to_client: flow.contains("to_client"),
                established: flow.contains("established"),
                stateless: flow.contains("stateless"),
            }
        })
    }

    fn parse_tcp_flags(options: &HashMap<String, String>) -> Option<TcpFlagsSpec> {
        options.get("flags").map(|flags| {
            let (modifier, flags) = if flags.starts_with('+') {
                (FlagModifier::All, &flags[1..])
            } else if flags.starts_with('*') {
                (FlagModifier::Any, &flags[1..])
            } else if flags.starts_with('!') {
                (FlagModifier::NotSet, &flags[1..])
            } else {
                (FlagModifier::Match, flags.as_str())
            };

            TcpFlagsSpec {
                flags: flags.to_string(),
                modifier,
            }
        })
    }

    fn classtype_to_severity(classtype: &str) -> Severity {
        match classtype {
            "attempted-admin" | "successful-admin" | "shellcode-detect" => Severity::Critical,
            "attempted-user" | "successful-user" | "trojan-activity" => Severity::High,
            "web-application-attack" | "attempted-dos" => Severity::Medium,
            "misc-activity" | "network-scan" | "not-suspicious" => Severity::Low,
            _ => Severity::Medium,
        }
    }
}

/// Rule matching engine
pub struct RuleEngine {
    rules: Vec<Rule>,
    fast_pattern_index: HashMap<Vec<u8>, Vec<usize>>,
}

impl RuleEngine {
    pub async fn load(
        rules_dirs: &[std::path::PathBuf],
        config: &IdsConfig,
    ) -> Result<Self, IdsError> {
        let mut rules = Vec::new();

        for dir in rules_dirs {
            if !dir.exists() {
                tracing::warn!("Rules directory not found: {}", dir.display());
                continue;
            }

            let mut entries = tokio::fs::read_dir(dir).await?;
            while let Some(entry) = entries.next_entry().await? {
                let path = entry.path();
                if path.extension().map(|e| e == "rules").unwrap_or(false) {
                    let content = tokio::fs::read_to_string(&path).await?;

                    for line in content.lines() {
                        match RuleParser::parse(line) {
                            Ok(rule) => rules.push(rule),
                            Err(IdsError::SkippedLine) => continue,
                            Err(e) => {
                                tracing::warn!("Failed to parse rule: {}", e);
                            }
                        }
                    }
                }
            }
        }

        // Build fast pattern index
        let fast_pattern_index = Self::build_fast_pattern_index(&rules);

        tracing::info!("Loaded {} IDS rules", rules.len());

        Ok(Self {
            rules,
            fast_pattern_index,
        })
    }

    fn build_fast_pattern_index(rules: &[Rule]) -> HashMap<Vec<u8>, Vec<usize>> {
        let mut index: HashMap<Vec<u8>, Vec<usize>> = HashMap::new();

        for (idx, rule) in rules.iter().enumerate() {
            for content in &rule.content_matches {
                if content.fast_pattern || content.pattern.len() >= 4 {
                    // Use first 4 bytes as index key
                    let key = if content.pattern.len() >= 4 {
                        content.pattern[..4].to_vec()
                    } else {
                        content.pattern.clone()
                    };
                    index.entry(key).or_default().push(idx);
                }
            }
        }

        index
    }

    pub async fn match_packet(&self, packet: &Packet, config: &IdsConfig) -> Vec<Rule> {
        let mut matched = Vec::new();

        // Use fast pattern index for initial filtering
        let candidate_indices: Vec<usize> = if !packet.payload.is_empty() {
            self.get_candidates_from_payload(&packet.payload)
        } else {
            (0..self.rules.len()).collect()
        };

        for &idx in &candidate_indices {
            let rule = &self.rules[idx];

            if self.matches_rule(rule, packet, config) {
                matched.push(rule.clone());
            }
        }

        matched
    }

    fn get_candidates_from_payload(&self, payload: &[u8]) -> Vec<usize> {
        let mut candidates = std::collections::HashSet::new();

        // Check 4-byte windows against fast pattern index
        for window in payload.windows(4) {
            if let Some(indices) = self.fast_pattern_index.get(window) {
                candidates.extend(indices);
            }
        }

        // If no fast pattern matches, return all rules
        if candidates.is_empty() {
            (0..self.rules.len()).collect()
        } else {
            candidates.into_iter().collect()
        }
    }

    fn matches_rule(&self, rule: &Rule, packet: &Packet, config: &IdsConfig) -> bool {
        // Check protocol
        if !self.matches_protocol(rule, packet) {
            return false;
        }

        // Check addresses
        if !self.matches_address(&rule.src_addr, &packet.src_ip, config) {
            return false;
        }
        if !self.matches_address(&rule.dst_addr, &packet.dst_ip, config) {
            return false;
        }

        // Check ports
        if !self.matches_port(&rule.src_port, packet.src_port) {
            return false;
        }
        if !self.matches_port(&rule.dst_port, packet.dst_port) {
            return false;
        }

        // Check TCP flags if specified
        if let Some(ref flags_spec) = rule.flags {
            if !self.matches_tcp_flags(flags_spec, &packet.flags) {
                return false;
            }
        }

        // Check content matches
        for content in &rule.content_matches {
            let matches = self.matches_content(content, &packet.payload);
            if content.negated {
                if matches {
                    return false;
                }
            } else if !matches {
                return false;
            }
        }

        // Check PCRE matches
        for pcre in &rule.pcre_matches {
            let payload_str = String::from_utf8_lossy(&packet.payload);
            let matches = pcre.pattern.is_match(&payload_str);
            if pcre.negated {
                if matches {
                    return false;
                }
            } else if !matches {
                return false;
            }
        }

        true
    }

    fn matches_protocol(&self, rule: &Rule, packet: &Packet) -> bool {
        match rule.protocol {
            RuleProtocol::Any => true,
            RuleProtocol::TCP => packet.protocol == Protocol::TCP,
            RuleProtocol::UDP => packet.protocol == Protocol::UDP,
            RuleProtocol::ICMP => packet.protocol == Protocol::ICMP,
            RuleProtocol::IP => true,
            RuleProtocol::HTTP => {
                packet.protocol == Protocol::TCP &&
                (packet.dst_port == 80 || packet.dst_port == 8080)
            }
            RuleProtocol::DNS => {
                packet.dst_port == 53 || packet.src_port == 53
            }
            RuleProtocol::TLS => {
                packet.protocol == Protocol::TCP && packet.dst_port == 443
            }
        }
    }

    fn matches_address(&self, spec: &AddressSpec, addr: &IpAddr, config: &IdsConfig) -> bool {
        match spec {
            AddressSpec::Any => true,
            AddressSpec::HomeNet => {
                config.home_net.iter().any(|net| net.contains(addr))
            }
            AddressSpec::ExternalNet => {
                !config.home_net.iter().any(|net| net.contains(addr))
            }
            AddressSpec::Address(net) => net.contains(addr),
            AddressSpec::AddressList(nets) => {
                nets.iter().any(|net| net.contains(addr))
            }
            AddressSpec::Negated(inner) => {
                !self.matches_address(inner, addr, config)
            }
        }
    }

    fn matches_port(&self, spec: &PortSpec, port: u16) -> bool {
        match spec {
            PortSpec::Any => true,
            PortSpec::Single(p) => *p == port,
            PortSpec::Range(start, end) => port >= *start && port <= *end,
            PortSpec::List(ports) => ports.contains(&port),
            PortSpec::Negated(inner) => !self.matches_port(inner, port),
        }
    }

    fn matches_tcp_flags(&self, spec: &TcpFlagsSpec, flags: &PacketFlags) -> bool {
        let flag_present = |c: char| -> bool {
            match c {
                'S' => flags.syn,
                'A' => flags.ack,
                'F' => flags.fin,
                'R' => flags.rst,
                'P' => flags.psh,
                'U' => flags.urg,
                _ => false,
            }
        };

        match spec.modifier {
            FlagModifier::Match => {
                spec.flags.chars().all(|c| flag_present(c))
            }
            FlagModifier::All => {
                spec.flags.chars().all(|c| flag_present(c))
            }
            FlagModifier::Any => {
                spec.flags.chars().any(|c| flag_present(c))
            }
            FlagModifier::NotSet => {
                spec.flags.chars().all(|c| !flag_present(c))
            }
        }
    }

    fn matches_content(&self, content: &ContentMatch, payload: &[u8]) -> bool {
        let pattern = if content.nocase {
            content.pattern.to_ascii_lowercase()
        } else {
            content.pattern.clone()
        };

        let search_payload = if content.nocase {
            payload.to_ascii_lowercase()
        } else {
            payload.to_vec()
        };

        // Apply offset/depth constraints
        let start = content.offset.unwrap_or(0);
        let end = content.depth
            .map(|d| (start + d).min(search_payload.len()))
            .unwrap_or(search_payload.len());

        if start >= search_payload.len() {
            return false;
        }

        let search_range = &search_payload[start..end];

        search_range.windows(pattern.len())
            .any(|window| window == pattern.as_slice())
    }
}

/// Alert manager for logging and rate limiting
pub struct AlertManager {
    log_file: tokio::sync::Mutex<tokio::fs::File>,
    rate_limiter: RateLimiter,
}

impl AlertManager {
    pub async fn new(
        log_path: &std::path::Path,
        max_alerts_per_second: u32,
    ) -> Result<Self, IdsError> {
        if let Some(parent) = log_path.parent() {
            tokio::fs::create_dir_all(parent).await?;
        }

        let log_file = tokio::fs::OpenOptions::new()
            .create(true)
            .append(true)
            .open(log_path)
            .await?;

        Ok(Self {
            log_file: tokio::sync::Mutex::new(log_file),
            rate_limiter: RateLimiter::new(max_alerts_per_second),
        })
    }

    pub async fn record_alert(&self, alert: &Alert) -> Result<(), IdsError> {
        // Rate limit alerts
        if !self.rate_limiter.check(alert.src_ip).await {
            return Ok(());
        }

        // Serialize alert to JSON
        let json = serde_json::to_string(&AlertLog::from(alert))
            .map_err(|e| IdsError::SerializationError(e.to_string()))?;

        // Write to log file
        use tokio::io::AsyncWriteExt;
        let mut file = self.log_file.lock().await;
        file.write_all(json.as_bytes()).await?;
        file.write_all(b"\n").await?;
        file.flush().await?;

        Ok(())
    }
}

/// IP blocker for IPS mode
pub struct Blocker {
    blocked: HashMap<IpAddr, BlockEntry>,
    threshold: u32,
    duration: std::time::Duration,
}

struct BlockEntry {
    blocked_at: std::time::Instant,
    alert_count: u32,
}

impl Blocker {
    pub fn new(threshold: u32, duration_seconds: u64) -> Self {
        Self {
            blocked: HashMap::new(),
            threshold,
            duration: std::time::Duration::from_secs(duration_seconds),
        }
    }

    pub fn is_blocked(&self, ip: &IpAddr) -> bool {
        if let Some(entry) = self.blocked.get(ip) {
            entry.blocked_at.elapsed() < self.duration
        } else {
            false
        }
    }

    pub fn add_block(&mut self, ip: &IpAddr) {
        let entry = self.blocked.entry(*ip).or_insert(BlockEntry {
            blocked_at: std::time::Instant::now(),
            alert_count: 0,
        });

        entry.alert_count += 1;

        if entry.alert_count >= self.threshold {
            entry.blocked_at = std::time::Instant::now();
            tracing::warn!("Blocked IP {} after {} alerts", ip, entry.alert_count);
        }
    }

    pub fn remove_block(&mut self, ip: &IpAddr) {
        self.blocked.remove(ip);
    }

    pub fn cleanup_expired(&mut self) {
        self.blocked.retain(|_, entry| {
            entry.blocked_at.elapsed() < self.duration
        });
    }
}
