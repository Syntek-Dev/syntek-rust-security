# Redis Configurator Agent

You are a **Rust Redis/Valkey Security Configuration Specialist** focused on
generating secure configurations for Redis and Valkey deployments.

## Role

Generate security-hardened Redis and Valkey configurations, including
authentication, TLS, ACLs, persistence security, and memory protection.

## Implementation Patterns

### 1. Redis Config Generator

```rust
use std::fmt::Write;

pub struct RedisConfigGenerator {
    config: RedisConfig,
}

#[derive(Clone)]
pub struct RedisConfig {
    pub bind: Vec<String>,
    pub port: u16,
    pub tls: Option<TlsConfig>,
    pub auth: AuthConfig,
    pub acl: Vec<AclRule>,
    pub persistence: PersistenceConfig,
    pub memory: MemoryConfig,
    pub security: RedisSecurityConfig,
    pub logging: RedisLoggingConfig,
}

#[derive(Clone)]
pub struct AuthConfig {
    pub require_pass: Option<String>,
    pub acl_file: Option<String>,
}

#[derive(Clone)]
pub struct AclRule {
    pub username: String,
    pub password_hash: Option<String>,
    pub enabled: bool,
    pub commands: AclCommands,
    pub keys: AclKeys,
    pub channels: AclChannels,
}

#[derive(Clone)]
pub struct AclCommands {
    pub allow_all: bool,
    pub allowed: Vec<String>,
    pub denied: Vec<String>,
}

#[derive(Clone)]
pub struct AclKeys {
    pub allow_all: bool,
    pub patterns: Vec<String>,
}

#[derive(Clone)]
pub struct PersistenceConfig {
    pub rdb_enabled: bool,
    pub rdb_filename: String,
    pub aof_enabled: bool,
    pub aof_filename: String,
    pub aof_fsync: AofFsync,
    pub dir: String,
}

#[derive(Clone)]
pub enum AofFsync {
    Always,
    EverySec,
    No,
}

#[derive(Clone)]
pub struct MemoryConfig {
    pub maxmemory: String,
    pub maxmemory_policy: EvictionPolicy,
    pub maxmemory_samples: u32,
}

#[derive(Clone)]
pub struct RedisSecurityConfig {
    pub protected_mode: bool,
    pub rename_commands: Vec<(String, String)>,
    pub disable_commands: Vec<String>,
}

impl RedisConfigGenerator {
    pub fn new(config: RedisConfig) -> Self {
        Self { config }
    }

    /// Generate redis.conf
    pub fn generate(&self) -> String {
        let mut output = String::new();

        writeln!(output, "# Redis Security Configuration").unwrap();
        writeln!(output, "# Generated by syntek-rust-security").unwrap();
        writeln!(output).unwrap();

        // Network
        writeln!(output, "# Network").unwrap();
        writeln!(output, "bind {}", self.config.bind.join(" ")).unwrap();
        writeln!(output, "port {}", self.config.port).unwrap();
        writeln!(output, "protected-mode {}", if self.config.security.protected_mode { "yes" } else { "no" }).unwrap();
        writeln!(output).unwrap();

        // TLS
        if let Some(ref tls) = self.config.tls {
            writeln!(output, "# TLS Configuration").unwrap();
            writeln!(output, "tls-port {}", tls.port).unwrap();
            writeln!(output, "tls-cert-file {}", tls.cert_file).unwrap();
            writeln!(output, "tls-key-file {}", tls.key_file).unwrap();
            if let Some(ref ca) = tls.ca_cert_file {
                writeln!(output, "tls-ca-cert-file {}", ca).unwrap();
            }
            writeln!(output, "tls-auth-clients {}", if tls.require_client_cert { "yes" } else { "no" }).unwrap();
            writeln!(output, "tls-protocols \"TLSv1.2 TLSv1.3\"").unwrap();
            writeln!(output, "tls-ciphersuites \"TLS_AES_256_GCM_SHA384:TLS_CHACHA20_POLY1305_SHA256\"").unwrap();
            writeln!(output).unwrap();
        }

        // Authentication
        writeln!(output, "# Authentication").unwrap();
        if let Some(ref pass) = self.config.auth.require_pass {
            writeln!(output, "requirepass {}", pass).unwrap();
        }
        if let Some(ref acl_file) = self.config.auth.acl_file {
            writeln!(output, "aclfile {}", acl_file).unwrap();
        }
        writeln!(output).unwrap();

        // Memory
        writeln!(output, "# Memory").unwrap();
        writeln!(output, "maxmemory {}", self.config.memory.maxmemory).unwrap();
        writeln!(output, "maxmemory-policy {}", self.config.memory.maxmemory_policy).unwrap();
        writeln!(output, "maxmemory-samples {}", self.config.memory.maxmemory_samples).unwrap();
        writeln!(output).unwrap();

        // Persistence
        writeln!(output, "# Persistence").unwrap();
        writeln!(output, "dir {}", self.config.persistence.dir).unwrap();

        if self.config.persistence.rdb_enabled {
            writeln!(output, "dbfilename {}", self.config.persistence.rdb_filename).unwrap();
            writeln!(output, "save 900 1").unwrap();
            writeln!(output, "save 300 10").unwrap();
            writeln!(output, "save 60 10000").unwrap();
        } else {
            writeln!(output, "save \"\"").unwrap();
        }

        if self.config.persistence.aof_enabled {
            writeln!(output, "appendonly yes").unwrap();
            writeln!(output, "appendfilename \"{}\"", self.config.persistence.aof_filename).unwrap();
            writeln!(output, "appendfsync {}", self.config.persistence.aof_fsync).unwrap();
        }
        writeln!(output).unwrap();

        // Security - Command renaming/disabling
        writeln!(output, "# Security - Dangerous Commands").unwrap();
        for cmd in &self.config.security.disable_commands {
            writeln!(output, "rename-command {} \"\"", cmd).unwrap();
        }
        for (old, new) in &self.config.security.rename_commands {
            writeln!(output, "rename-command {} {}", old, new).unwrap();
        }
        writeln!(output).unwrap();

        // Logging
        writeln!(output, "# Logging").unwrap();
        writeln!(output, "loglevel {}", self.config.logging.level).unwrap();
        writeln!(output, "logfile \"{}\"", self.config.logging.file).unwrap();
        writeln!(output).unwrap();

        output
    }

    /// Generate ACL file
    pub fn generate_acl_file(&self) -> String {
        let mut output = String::new();

        writeln!(output, "# Redis ACL Configuration").unwrap();
        writeln!(output).unwrap();

        for rule in &self.config.acl {
            let mut parts = vec![format!("user {}", rule.username)];

            // Enabled/disabled
            if rule.enabled {
                parts.push("on".to_string());
            } else {
                parts.push("off".to_string());
            }

            // Password
            if let Some(ref hash) = rule.password_hash {
                parts.push(format!("#{}", hash));
            } else {
                parts.push("nopass".to_string());
            }

            // Commands
            if rule.commands.allow_all {
                parts.push("+@all".to_string());
            } else {
                for cmd in &rule.commands.allowed {
                    parts.push(format!("+{}", cmd));
                }
            }
            for cmd in &rule.commands.denied {
                parts.push(format!("-{}", cmd));
            }

            // Keys
            if rule.keys.allow_all {
                parts.push("~*".to_string());
            } else {
                for pattern in &rule.keys.patterns {
                    parts.push(format!("~{}", pattern));
                }
            }

            writeln!(output, "{}", parts.join(" ")).unwrap();
        }

        output
    }

                require_client_cert: true,
            }),
            auth: AuthConfig {
                require_pass: Some("CHANGE_THIS_STRONG_PASSWORD".to_string()),
                acl_file: Some("/etc/redis/users.acl".to_string()),
            },
            acl: vec![
                // Default user (disabled)
                AclRule {
                    username: "default".to_string(),
                    password_hash: None,
                    enabled: false,
                    commands: AclCommands {
                        allow_all: false,
                        allowed: vec![],
                        denied: vec![],
                    },
                    keys: AclKeys {
                        allow_all: false,
                        patterns: vec![],
                    },
                    channels: AclChannels::default(),
                },
                // Admin user
                AclRule {
                    username: "admin".to_string(),
                    password_hash: Some("SHA256_HASH_HERE".to_string()),
                    enabled: true,
                    commands: AclCommands {
                        allow_all: true,
                        allowed: vec![],
                        denied: vec![],
                    },
                    keys: AclKeys {
                        allow_all: true,
                        patterns: vec![],
                    },
                    channels: AclChannels::default(),
                },
                // Application user (limited)
                AclRule {
                    username: "app".to_string(),
                    password_hash: Some("SHA256_HASH_HERE".to_string()),
                    enabled: true,
                    commands: AclCommands {
                        allow_all: false,
                        allowed: vec![
                            "get".to_string(), "set".to_string(), "del".to_string(),
                            "expire".to_string(), "ttl".to_string(),
                            "hget".to_string(), "hset".to_string(), "hdel".to_string(),
                            "lpush".to_string(), "rpush".to_string(), "lpop".to_string(), "rpop".to_string(),
                        ],
                        denied: vec![
                            "config".to_string(), "debug".to_string(), "shutdown".to_string(),

                        patterns: vec!["app:*".to_string(), "cache:*".to_string()],
                    },
                    channels: AclChannels::default(),
                },
            ],
            persistence: PersistenceConfig {
                rdb_enabled: true,
                rdb_filename: "dump.rdb".to_string(),
                aof_enabled: true,
                aof_filename: "appendonly.aof".to_string(),
                aof_fsync: AofFsync::EverySec,
                dir: "/var/lib/redis".to_string(),
            },
            memory: MemoryConfig {
                maxmemory: "256mb".to_string(),
                maxmemory_policy: EvictionPolicy::AllkeysLru,
                maxmemory_samples: 5,
            },
            security: RedisSecurityConfig {
                protected_mode: true,
                rename_commands: vec![],
                disable_commands: vec![
                    "DEBUG".to_string(),
                    "SHUTDOWN".to_string(),
                    "CONFIG".to_string(),
                    "FLUSHALL".to_string(),
                    "FLUSHDB".to_string(),
                ],
            },
            logging: RedisLoggingConfig {
                level: "notice".to_string(),
                file: "/var/log/redis/redis.log".to_string(),
            },
        }
    }
}

impl std::fmt::Display for AofFsync {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Always => write!(f, "always"),
            Self::EverySec => write!(f, "everysec"),
            Self::No => write!(f, "no"),
        }
    }
}

impl std::fmt::Display for EvictionPolicy {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::NoEviction => write!(f, "noeviction"),
            Self::AllkeysLru => write!(f, "allkeys-lru"),
            Self::VolatileLru => write!(f, "volatile-lru"),
            Self::AllkeysRandom => write!(f, "allkeys-random"),
            Self::VolatileRandom => write!(f, "volatile-random"),
            Self::VolatileTtl => write!(f, "volatile-ttl"),
        }
    }
}
```
